  • 打印二维数组 p *matrix@cols@rows
  • 打印二维数组 printMatrix matrix cols rows
    define printMatrix
  set $arr = $arg0
  set $rows = $arg1
  set $cols = $arg2
  set $i = 0
  printf "\n"
  while $i < $rows
      set $j = 0
      while $j < $cols
          printf "%02d, ",$arr[$i][$j]
          set $j = $j + 1
      end
      printf "\n"
      set $i = $i + 1
  end
end
